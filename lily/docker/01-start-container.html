# 启动常用的一些容器服务

# MySQL

docker run -p 3306:3306 -p 33060:33060 \
-v /data/docker/mysql/lib:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=000000 \
--restart=always \
--name mysql \
-d mysql:latest \
--lower_case_table_names=1
  • -v 参数是将主机磁盘目录挂载到容器中的指定目录
  • -e 在启动容器时设置一些程序的初始变量,这里是设置 root 密码
  • --restart=always 容器随 Docker 一同启动
  • --name mysql 设置启动容器的名称
  • -d mysql:latest 启动容器时要使用的镜像
  • --lower_case_table_names=1 设置 MySQL 大小写为不敏感,即关闭大小写敏感

可能需要的一些命令

# 在容器 mysql 中开启一个交互模式的终端
docker exec -it mysql /bin/bash
# 登录 mysql 数据库
mysql -uroot -p000000

GRANT ALL PRIVILEGES ON *.* TO 'root'@'localhost' IDENTIFIED BY '000000';
GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' IDENTIFIED BY '000000';

# Mariadb

docker run -p 3306:3306 \
-v /data/docker/mariadb:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=000000 \
--restart=always \
--name mariadb \
-d mariadb:latest \
--lower_case_table_names=1

同 MySql

# PostgreSQL

docker run -p 5432:5432 \
--restart=always \
-v /data/docker/postgresql:/var/lib/postgresql/data \
-e POSTGRES_PASSWORD=000000 \
--name=postgresql \
-d postgresql

# Redis

docker run -p 6379:6379 \
--restart=always \
-v /data/docker/redis/conf:/etc/redis \
-v /data/docker/redis/data:/data
--name redis \
-d redis \
redis-server --appendonly yes
  • redis-server --appendonly yes 在容器执行redis-server启动命令，并打开 redis 持久化配置
